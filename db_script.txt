CREATE TABLE tbl_users (
	user_id BIGSERIAL PRIMARY KEY,
	firstname VARCHAR(200) NOT NULL,
	lastname VARCHAR(200) NOT NULL,
	email VARCHAR(200) NOT NULL UNIQUE,
	password VARCHAR(200) NOT NULL,
	role_id INT REFERENCES tbl_role(role_id)
	
);


	  
CREATE TABLE tbl_role (
	role_id INT PRIMARY KEY,
	role_name VARCHAR ( 50 ),
	UNIQUE (role_id)
);


CREATE TABLE tbl_categories (
	cate_id INT  ,
	cate_name VARCHAR ( 50 )
	
);


CREATE TABLE tbl_movies (
	movie_id INT  ,
	movie_name VARCHAR ( 50 ),
	movie_desc VARCHAR ( 550 ),
	movie_cover VARCHAR ( 50 ),
	movie_link VARCHAR ( 50 ),
	movie_release_date VARCHAR ( 50 ),
	movie_view INT,
	cate_id INT,
	price_id INT,
	theater INT
	
);

CREATE TABLE tbl_price (
	price_id INT  ,
	movie_price VARCHAR ( 50 ),
	movie_id INT	
);

	
CREATE TABLE tbl_tickets (
	tickets_id INT  ,
	tickets_name VARCHAR ( 50 ),
	showtimes VARCHAR ( 50 ),
	discount VARCHAR ( 50 ),
	price_id INT,
	user_id INT,
	movie_id INT,
	cate_id INT,
	remark VARCHAR ( 50 )
);
